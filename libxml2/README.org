** Example1 - Converting XML File to S-expression tree
 
*** XML File 

 file:tests/cars.xml 

 #+BEGIN_SRC xml 
<?xml version="1.0" encoding="UTF-8" ?>
<cars>
  <car country="JP">
    <name>motorcycle</name>
    <price>150</price>
    <img file="car1.jpg"/>
  </car>
  <car country="US">
    <name>truck</name>
    <price>500</price>
    <img file="car2.jpg"/>
  </car>
  <car country="DE">
    <name>car</name>
    <price>200</price>
    <img file="car3.jpg"/>
  </car>
</cars> 
 #+END_SRC
